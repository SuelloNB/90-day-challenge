/*===============================================================
    PADDING EXAMPLES
  
    - Padding is the space between the content of an element and its border.
    - It creates space inside the element, pushing the content away from the edges.
================================================================ */
.padding-example p{
    padding-top: 10px; /* Adds 10px space above the content */
    padding-right: 20px; /* Adds 20px space to the right of the content */
    padding-bottom: 30px; /* Adds 30px space below the content */
    padding-left: 40px; /* Adds 40px space to the left of the content */
    border: 2px solid black; /* This is just to visualize the padding */
}
