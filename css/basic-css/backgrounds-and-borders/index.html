Backround Image Property


/*Background Image Property*/
.bg-img{
    background-image: url("../assets/images/sunflower.png"); /*Set a background image to a container*/
}
        

/*Background Size Property*/
.bg-img-size{
    background-image: url("../assets/images/sunflower.png"); /*Set a background image to a container*/
background-size: contain; /*set the image size to contain*/
}
